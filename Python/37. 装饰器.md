#### 自定义 装饰器

```python
import time
def timer(func):                 # 倒数第二层，永远是传递 func
    def inner(*args, **kwargs):  # 装饰后的方法
        start = time.time()
        res = func(*args, **kwargs)
        end = time.time()
        print(end - start)
        return res
    return inner

@timer   # 相当于 test = timer(test)
def test(n):
    for x in range(n):
        print(x)
```

#### 带参数的装饰器

在外面再嵌套了一层函数。

```python
def timer(data):
    def out(func):    # 倒数第二层，永远是传递 func
        def inner(): # 装饰后的方法 
            import time
            start = time.clock()
            func()
            end = time.clock()
            print(end - start)
            print(data)

        return inner

    return out


@timer(data="test")    # 相当于 test = timer(data)(test)
def test():
    for item in range(100000):
        pass


test()

```
